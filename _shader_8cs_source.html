231  private void GetParams(string source)
232  {
233  List<ProgramParam> shaderParams = new List<ProgramParam>();
234  var tokens = GLSLLexer.GetTokensFromMemory(source);
235 
236  for (int i = 0; i < tokens.Count; i++)
237  {
238  if (tokens[i].TokenType == GLSLLexer.TokenType.Keyword && (tokens[i].Text == "uniform" || tokens[i].Text == "attribute" || tokens[i].Text == "in"))
239  {
240  // get the parameter type (either uniform or attribute/in)
241  ParamType paramType = (tokens[i].Text == "uniform" ? ParamType.Uniform : ParamType.Attribute);
242  i++; // move past the uniform/attribute/in keyword
243 
244  // get the glsl type of the parameter
245  if (i == tokens.Count) break;
246  Type type = GlslTypeFromString(tokens[i].Text);
247  i++; // move past the type
248 
249  // now continue reading parameters until we hit EOF, semi-colon or the glsl programmer assigns a default value
250  while (i < tokens.Count && tokens[i].Text != ";")
251  {
252  if (tokens[i].TokenType == GLSLLexer.TokenType.Word) shaderParams.Add(new ProgramParam(type, paramType, tokens[i].Text));
253  else if (tokens[i].Text == "=") break; // they've assigned a default value, so continue on
254  i++;
255  }
256  }
257  }
258 
259  this.ShaderParams = shaderParams.ToArray();
260  }

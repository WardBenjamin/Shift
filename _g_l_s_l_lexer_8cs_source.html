120  public static TokenType GetTokenType(Match m)
121  {
122  if (m.Groups["word"].Success)
123  {
124  if (keywordDictionary.ContainsKey(m.Value)) return TokenType.Keyword;
125  else return TokenType.Word;
126  }
127  else if (m.Groups["number"].Success) return TokenType.Number;
128  else if (m.Groups["operator"].Success) return TokenType.Operator;
129  else if (m.Groups["string"].Success) return TokenType.String;
130  else if (m.Groups["syntax"].Success) return TokenType.Syntax;
131  else return TokenType.Unknown;

179  public static List<Token> GetTokens(string line, int lineNumber = 0, int offset = 0)
187  {
188  List<Token> tokens = new List<Token>();
189  MatchCollection matches = Regex.Matches(line, pattern);
190 
191  foreach (Match m in matches)
192  {
193  if (m.Value == " " || m.Value == "\t") continue;
194  tokens.Add(new Token(m.Value, lineNumber, offset + m.Index, GetTokenType(m)));
195  }

205  public static List<Token> GetTokens(List<Line> lines)
206  {
207  List<Token> tokens = new List<Token>();
208  List<Token> todos = new List<Token>();
209  List<Token> comments = new List<Token>();
210 
211  for (int i = 0; i < lines.Count; i++)
212  {
213  if (lines[i].Comment)
214  {
215  if (lines[i].Text.ToUpper().Contains("TODO:"))
216  {
217  string line = lines[i].Text;
218  while (line.Contains("//"))
219  {
220  if (line.IndexOf("//") < line.ToUpper().IndexOf("TODO:")) line = line.Substring(line.IndexOf("//") + 2);
221  else break;
222  }
223  todos.Add(new Token(line.Trim(), lines[i].LineNumber, 0, TokenType.TODO));
224  }
225  else comments.Add(new Token(lines[i].Text, lines[i].LineNumber, 0, TokenType.Comment));
226  }
227  else tokens.AddRange(GetTokens(lines[i].Text, lines[i].LineNumber));
228  }
229 
230  tokens.AddRange(todos); // add the todos to the end of the token list
231  tokens.AddRange(comments);
232 
233  return tokens;
234  }

280  private static List<Line> GetLinesFromMemoryInternal(string fileContents, string filename = "")
281  {
282  List<Line> lines = new List<Line>();
283  int lineNumber = 0;
284 
285  using (StringReader stream = new StringReader(fileContents))
286  {
287  bool comment = false, remove = false;
288  string temp = "";
289 
290  while ((temp = stream.ReadLine()) != null)
291  {
292  lineNumber++;
293 
294  // remove // comments first, they have priority
295  if (temp.Contains("//"))
296  {
297  int index = temp.IndexOf("//");
298 
299  // watch out for *///
300  if (index > 0 && temp[index - 1] == '*')
301  {
302  string sub = temp.Substring(index + 1);
303  if (sub.Contains("//")) index = temp.IndexOf("//");
304  else index = -1;
305  }
306 
307  if (index != -1)
308  {
309  if (!remove) lines.Add(new Line(temp.Substring(index + 2), lineNumber, true));
310  temp = temp.Substring(0, index);
311  }
312  }
313 
314  // search for the end of a comment block if we're currently in one
315  if (comment)
316  {
317  if (temp.Contains("*/")) { temp = temp.Substring(temp.IndexOf("*/") + 2); comment = false; }
318  else continue;
319  }
320 
321  // search for inline comment blocks
322  string line = temp;
323  while (line.Contains("/*"))
324  {
325  line = line.Substring(0, line.IndexOf("/*"));
326  comment = true;
327  if (temp.Contains("*/"))
328  {
329  temp = temp.Substring(temp.IndexOf("*/") + 2);
330  line += temp;
331  comment = false;
332  }
333  }
334 
335  if (!remove)
336  {
337  if (line.Length > 0) lines.Add(new Line(line, lineNumber));
338  }
339  }
340  }
341 
342  return lines;
343  }
